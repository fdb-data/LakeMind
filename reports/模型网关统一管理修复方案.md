# 模型网关统一管理 — 修复与重构方案

> 基于 `reports/v0.2.0的模型网关以及配套配置的设计开发方案总结.md` 的问题分析，
> 本文件给出 **可执行的修复方案**，满足以下需求：
>
> 1. ModelServing 统一管理
> 2. 用户可动态配置、加载本地离线模型（管理员放置文件到指定目录）
> 3. 可快速注册和发布外部模型 API
> 4. 不重启容器即可生效（热加载）
> 5. 修复 bug 和硬编码问题

---

## 1. 问题根因

当前模型网关有 **两套并行注册表**，互不同步：

| 系统 | 表 | 数据来源 | 影响运行时路由? |
|------|-----|---------|----------------|
| ModelServing | `model_registry`（0 行） | yaml 启动加载 | **是**（litellm Router） |
| Server | 5 表（21 行） | bootstrap + UI | **否**（纯展示） |

**根因**：控制平面（Server 5 表 + UI）写入的配置不传播到数据平面（ModelServing litellm Router）。UI 是展示性的，真正改路由需改 yaml + 重启。

**修复核心思路**：让 **ModelServing 同时拥有控制平面和数据平面** — DB 是单一真相源，API 写 DB → 同步热加载到 litellm Router / 本地模型字典，无需重启。

### 4.2 EmbeddingManager（多模型）

```python
class EmbeddingManager:
    """管理多个本地 fastembed 模型，按需 lazy load。"""

    def __init__(self, cache_dir: str = "/data/fastembed_cache"):
        self._cache_dir = cache_dir
        self._models: dict[str, TextEmbedding] = {}   # name → loaded model
        self._dims: dict[str, int] = {}                # name → dim
        self._lock = threading.Lock()

    def register(self, name: str, dim: int, model_path: str | None = None):
        """注册模型（不立即加载，首次 embed 时 lazy load）。"""
        with self._lock:
            self._dims[name] = dim

    def unregister(self, name: str):
        """卸载模型，释放内存。"""
        with self._lock:
            self._models.pop(name, None)
            self._dims.pop(name, None)

    def embed(self, texts: list[str], model_name: str) -> tuple[list[list[float]], int]:
        """嵌入文本，首次调用时 lazy load。"""
        if model_name not in self._models:
            with self._lock:
                if model_name not in self._models:
                    from fastembed import TextEmbedding
                    self._models[model_name] = TextEmbedding(
                        model_name=model_name, cache_dir=self._cache_dir)
        model = self._models[model_name]
        vectors = [[float(x) for x in v] for v in model.embed(texts)]
        return vectors, self._dims.get(model_name, 0)
```

### 4.4 Registry 重写（2 表 CRUD + 热加载核心）

```python
class ModelRegistry:
    """ms_models + ms_model_profiles 两表 CRUD，写 DB 后同步热加载。"""

    def __init__(self, dsn, gateway, embedding_mgr, asr_mgr):
        self._gateway = gateway
        self._embedding_mgr = embedding_mgr
        self._asr_mgr = asr_mgr

    def create_model(self, **kwargs) -> dict:
        """创建模型 → 写 DB → 如果 enabled 则热加载。"""
        model_id = _ulid("mdl")
        execute("INSERT INTO ms_models ...", (model_id, ...))
        model = self.get_model(model_id)
        if model["status"] == "enabled":
            self._hot_load(model)
        return model

    def enable_model(self, model_id: str) -> dict:
        """启用 → 写 DB → 热加载。"""
        execute("UPDATE ms_models SET status='enabled' ...", ...)
        model = self.get_model(model_id)
        self._hot_load(model)
        return model

    def disable_model(self, model_id: str) -> dict:
        """禁用 → 写 DB → 热卸载。"""
        execute("UPDATE ms_models SET status='disabled' ...", ...)
        self._hot_unload(model_id)
        return self.get_model(model_id)

    # ── 热加载核心 ──

    def _hot_load(self, model: dict):
        """根据 model_type + source 热加载到对应引擎。"""
        if model["model_type"] == "chat":
            self._gateway.register_model({
                "model_id": model["name"],
                "litellm_model": model["litellm_model"],
                "api_key": model["api_key"],
                "base_url": model["base_url"],
            })
        elif model["model_type"] == "embedding" and model["source"] == "local":
            self._embedding_mgr.register(
                name=model["name"], dim=model["embedding_dim"],
                model_path=model["model_path"])
        elif model["model_type"] == "asr" and model["source"] == "local":
            self._asr_mgr.register(
                model_id=model["name"], model_path=model["model_path"],
                config=model["model_config"])
        elif model["source"] == "external":
            self._gateway.register_model({...})

    def _hot_unload(self, model_id: str):
        """从引擎中卸载。"""
        model = self.get_model(model_id)
        if not model: return
        if model["model_type"] == "chat":
            self._gateway.deregister_model(model["name"])
        elif model["model_type"] == "embedding" and model["source"] == "local":
            self._embedding_mgr.unregister(model["name"])
        elif model["model_type"] == "asr" and model["source"] == "local":
            self._asr_mgr.unregister(model["name"])

    def load_all_enabled(self):
        """启动时从 DB 加载所有 enabled 模型。"""
        models = execute("SELECT * FROM ms_models WHERE status = 'enabled'")
        for model in models:
            try: self._hot_load(model)
            except Exception as e: logger.error(...)

    def resolve_profile(self, profile_name: str, tenant_id: str = None) -> dict:
        """解析 profile → 返回实际模型配置。"""
        row = execute_one("""
            SELECT p.*, m.name AS model_name, m.model_type, ...
            FROM ms_model_profiles p
            JOIN ms_models m ON p.model_id = m.model_id
            WHERE p.name = %s AND m.status = 'enabled'
        """, (profile_name,))
        if not row: raise ValueError(f"PROFILE_NOT_FOUND: {profile_name}")
        return row
```

### 4.6 启动流程

```python
def create_app() -> FastAPI:
    cfg = load_config()

    # 1. 创建管理器（空）
    embedding_mgr = EmbeddingManager(cache_dir=cfg.embedding.built_in.cache_dir)
    asr_mgr = ASRManager()
    gateway = ModelGateway(providers=[], gateway_config={...})  # 空 Router

    # 2. Registry
    registry = ModelRegistry(dsn=..., gateway=gateway,
                             embedding_mgr=embedding_mgr, asr_mgr=asr_mgr)

    # 3. DB 为空时从 yaml seed（仅首次启动）
    if registry.is_empty():
        registry.seed_from_yaml(cfg)

    # 4. 从 DB 加载所有 enabled 模型
    registry.load_all_enabled()

    # 5. 挂到 app.state
    app.state.gateway = gateway
    app.state.embedding_mgr = embedding_mgr
    app.state.asr_mgr = asr_mgr
    app.state.registry = registry
    return app
```

**关键**：litellm Router 启动时为空，从 DB 加载。yaml 只在 DB 为空时 seed。后续所有变更通过 API → DB → 热加载。

@app.api_route("/models/{path:path}", methods=["GET", "POST", "PUT", "DELETE"])
async def models_proxy(request: Request, path: str = ""):
    ms_url = os.environ["MODEL_SERVING_URL"]  # http://lakemind-model-serving:10824
    ms_key = os.environ["MODELSERVING_API_KEY"]
    async with httpx.AsyncClient() as client:
        resp = await client.request(
            request.method, f"{ms_url}/v1/{path}",
            headers={"Authorization": f"Bearer {ms_key}"},
            content=await request.body(),
            params=request.query_params,
        )
        return JSONResponse(content=resp.json(), status_code=resp.status_code)

## 11. 总结

| 维度 | 之前 | 之后 |
|------|------|------|
| 注册表 | 6 表（2 套，不同步） | 2 表（1 套，单一真相源） |
| 管理 API | Server 16 端点（展示性） | ModelServing 12 端点（热加载） |
| 配置生效 | 改 yaml + 重启 | API → DB → 热加载 |
| 本地模型 | 单模型，启动加载 | 多模型，lazy load |
| 消费者 | 硬编码模型名 | Profile 名（运行时可切换） |
| 代码量 | ~2426 行 | ~1800 行（删 Server 693 行，ModelServing 重写） |
| UI | 3 Tab + 4 Modal | 2 Tab + 2 Modal |

**核心原则**：**谁路由，谁管理**。ModelServing 是唯一知道如何路由请求的组件，因此它也应该是唯一管理路由配置的组件。Server 退化为纯数据平面，不再参与模型管理。
